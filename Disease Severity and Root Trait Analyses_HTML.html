Function to Analyze and Visualize Trait Data ```{r analysis-function} analyze_and_plot <- function(data, trait_col, trait_label, y_label, color = "steelblue", y_offset = 0.05) { summary_df <- data %>% group_by(treaments) %>% summarise( Mean = mean(.data[[trait_col]], na.rm = TRUE), SD = sd(.data[[trait_col]], na.rm = TRUE), n = n(), SE = SD / sqrt(n), .groups = "drop" ) formula <- as.formula(paste(trait_col, "~ treaments")) anova_model <- aov(formula, data = data) anova_table <- summary(anova_model) print(anova_table) write.csv(as.data.frame(anova_table[[1]]), paste0(trait_col, "_ANOVA_table.csv")) tukey_result <- TukeyHSD(anova_model) tukey_letters <- multcompLetters(tukey_result$treaments[,"p adj"])$Letters summary_df$group <- tukey_letters[match(summary_df$treaments, names(tukey_letters))] write.csv(as.data.frame(tukey_result$treaments), paste0(trait_col, "_TukeyHSD.csv")) write.csv(summary_df, paste0(trait_col, "_summary.csv"), row.names = FALSE) bar_plot <- ggplot(summary_df, aes(x = treaments, y = Mean, fill = treaments)) + geom_col(color = "black", fill = color, width = 0.7) + geom_errorbar(aes(ymin = Mean - SE, ymax = Mean + SE), width = 0.2) + geom_text(aes(label = group, y = Mean + SE + y_offset), size = 5) + labs( title = paste("Mean", trait_label, "by Treatment"), x = "Treatment", y = y_label, caption = "Different letters indicate significant differences (Tukey HSD, p < 0.05)" ) + annotate("text", x = 1, y = max(summary_df$Mean + summary_df$SE) + y_offset * 3, label = "Treatments sharing a letter are not significantly different", hjust = 0, size = 4) + theme_minimal() box_plot <- ggplot(data, aes(x = treaments, y = .data[[trait_col]], fill = treaments)) + geom_boxplot(alpha = 0.7) + labs( title = paste("Distribution of", trait_label, "by Treatment"), x = "Treatment", y = y_label ) + theme_minimal() list(bar_plot = bar_plot, box_plot = box_plot) } ``` ## 4.
